\paragraph{Exercice 1.} Les fonctions suivantes sont-elles affines ? Si oui,
identifier les coefficents $m$ et $p$.
\begin{align*}
  \textbf{a)}\;& f(x) = 5x-3 &
  \textbf{b)}\;& g(x) = x^2-1 &
  \textbf{c)}\;& h(x) = 9-x &
  \textbf{d)}\;& t(x) = \frac{7-4x}{6}
\end{align*}
